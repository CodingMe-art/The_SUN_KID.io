ចលនាធំ និង ចលនាតូច — សម្រាប់កុមារពិសេស

ឯកសារនេះមានសកម្មភាព យុទ្ធសាស្ត្រ និងតារាងតាមដាន សមរម្យសម្រាប់កុមារដែលមានតម្រូវការាពិសេស (Autism, Down syndrome, ADHD, Developmental Delay ...)

✋ ចលនាតូច (Fine Motor)

ចលនាតូចទាក់ទងនឹង សាច់ដុំតូចៗ — ម្រាមដៃ បបូរមាត់។ វាជាជំនាញសម្រាប់សរសេរ កាត់ កាន់ និងបញ្ចូលវត្ថុតូចៗ។

បញ្ហាធម្មតា:
  • ចាប់ខ្មៅដៃបានមិនច្បាស់
  • ការសរសេរ និងកាត់ក្រដាសលំបាក
  • ការសម្របសម្រួលភ្នែក-ដៃខ្សោយ

🦵 ចលនាធំ (Gross Motor)

ចលនាធំទាក់ទងនឹងសាច់ដុំធំៗ — ដើរ រត់ លោត និងតុល្យភាព។ វាជាសមត្ថភាពសម្រាប់សុខភាព និងចលនាទីលាន។

បញ្ហាធម្មតា:
  • តុល្យភាពខ្សោយ ងាយដួល
  • កម្លាំងជើង/ដៃអន់
  • ការចូលរួមក្នុងល្បែងក្រុមមានអាក្រក់

សកម្មភាពជាក់លាក់ (Activity Ideas)

🧩
Lego ឬប៊្លុកធំ (Begin with large pieces)

ធ្វើជំហានតូចៗ: រួមគ្នាធ្វើម៉ូឌែល ២–៣ ដុំ → ចាប់ដាក់ → បកដោះ។ ជួយលើកសមត្ថភាពចាប់ និងបញ្ចូល។

✂️
កាត់ក្រដាសដោយកន្ត្រៃ (Scissor skills)

ចាប់ផ្តើមជាមួយក្រដាសធន់ និងកន្ត្រៃមានសុវត្ថិភាព។ សិក្សាផ្នែកលើ-ក្រោមជំហាន និងទទួលស្គាល់សញ្ញាព្រមទាំងពិន្ទុ។

🎯
បោះបាល់តូច / ទម្លាក់គ្រាប់

ប្រើបាល់ខ្នាតធំឬធន់ដើម្បីលើកកំហែងបោះ និងខ្សោយ—ជួយសម្របសម្រួលដៃ-ភ្នែក។

🍽️
សំរាប់ភាពឯករាជ្យ (Feeding practice)

ប្រើស្លាបព្រា និងពែងធំមុន → បង្រៀនថ្នាក់បបូរមាត់ → ជួយទទួលខុសត្រូវ។

តារាងតាមដានសមត្ថភាព (Progress Tracker)

ទុកសំគាល់ប្រចាំសប្តាហ៍/ខែ — ពិនិត្យ និងខ្សែតាមដានជំហាននិទាឃរដូវ

សម្គាល់: ប្រើកំណត់ថ្ងៃជាក់លាក់ (ឧ. 2025-11-05) និងចុះតុល្យភាព/ដំណើរការ។

យុទ្ធសាស្ត្រសម្រាប់អ្នកបង្រៀន និងឪពុកម្តាយ

  • បំបែកកិច្ចការជាជំហានតូចៗ និងប្រើ visual cue (រូបភាព/សញ្ញា)
  • ប្រើសម្ភារៈធំៗជាមុនសម្រាប់ទម្លាប់ការចាប់
  • ផ្តល់ពង្រឹងយ៉ាងឆាប់និងច្រើន (praise & small rewards)
  • ព្យាយាមធ្វើវាជាការលេង — មិនចង្អៀតកុមារ
  • ធ្វើការត្រួតពិនិត្យបន្តិចៗ ហើយកំណត់គោលដៅតិចៗដែលអាចសម្រេចបាន

Quick adaptations

Use adaptive tools: larger pencils, textured grips, non-slip mats, weighted cutlery, and supportive seating (wedge cushions).

ព្រឹត្តិប័ត្រ៖ សកម្មភាពសម្រាប់ប្រើជារៀងរាល់សប្តាហ៍

សកម្មភាព លក្ខណៈ គោលដៅ កម្រិតងាយ
ចាប់ប៊្លុក ធំ → តូច ចាប់, ដាក់, បញ្ចេញ ពង្រឹង pinch និង coordination ងាយ → មធ្យម
កាត់ក្រដាសដាមគន្លង កាត់, តាមបន្ទាត់ ប្រើកន្ត្រៃ និងដៃដែលអាចគ្រប់គ្រង មធ្យម
បោះបាល់ទៅគោលដៅ បោះ, គោលដៅ coordination & targeting ងាយ
ដើរលើបន្ទាត់ (balance beam) តុល្យភាព balance & core stability មធ្យម → ពិបាក
